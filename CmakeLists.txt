cmake_minimum_required(VERSION 3.20)
project(WyvernServer LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(POLICY CMP0141)
  cmake_policy(SET CMP0141 NEW)
endif()
if(POLICY CMP0167)
  cmake_policy(SET CMP0167 OLD)
endif()

option(WYVERN_BUILD_FOUNDATION "Build Wyvern foundation server target" ON)

set(WYVERN_CORE_DIR "" CACHE PATH "Path to external Core directory")
set(_wyvern_core_candidates "")
if(WYVERN_CORE_DIR)
  list(APPEND _wyvern_core_candidates "${WYVERN_CORE_DIR}")
endif()
list(APPEND _wyvern_core_candidates
  "${CMAKE_SOURCE_DIR}/../Core"
  "${CMAKE_SOURCE_DIR}/Core"
)

set(WYVERN_RESOLVED_CORE_DIR "")
foreach(_candidate IN LISTS _wyvern_core_candidates)
  if(EXISTS "${_candidate}/CmakeLists.txt" OR EXISTS "${_candidate}/CMakeLists.txt")
    set(WYVERN_RESOLVED_CORE_DIR "${_candidate}")
    break()
  endif()
endforeach()

if(NOT WYVERN_RESOLVED_CORE_DIR)
  message(FATAL_ERROR
    "Core was not found. Set WYVERN_CORE_DIR to external Core path "
    "or keep Core as ../Core (preferred) / Core (fallback)."
  )
endif()

message(STATUS "Using Core from: ${WYVERN_RESOLVED_CORE_DIR}")
add_subdirectory("${WYVERN_RESOLVED_CORE_DIR}" "${CMAKE_BINARY_DIR}/_deps/wyvern_core")

if(WYVERN_BUILD_FOUNDATION)
  find_package(Boost REQUIRED COMPONENTS system thread)

  add_executable(wyvern_server_foundation
    Server/platform/ServerFoundationMain.cpp
    Server/platform/PlatformModules.cpp
    Server/server/FileCache.cpp
    Server/server/RequestHandler.cpp
  )

  target_include_directories(wyvern_server_foundation
    PRIVATE
      ${CMAKE_SOURCE_DIR}/Server/platform
      ${CMAKE_SOURCE_DIR}/Server/server
      ${CMAKE_SOURCE_DIR}/Server/utils
      ${CMAKE_SOURCE_DIR}/Server/modules/Transport
  )

  target_link_libraries(wyvern_server_foundation
    PRIVATE
      WyvernCore
      Boost::system
      Boost::thread
  )

  if(WIN32)
    target_compile_definitions(wyvern_server_foundation PRIVATE _WIN32_WINNT=0x0A00 WIN32_LEAN_AND_MEAN NOMINMAX)
    target_link_libraries(wyvern_server_foundation PRIVATE ws2_32 mswsock)
  endif()
endif()
