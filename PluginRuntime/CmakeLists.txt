cmake_minimum_required(VERSION 3.20)
project(WyvernPluginRuntime LANGUAGES CXX)

add_library(wyvern_plugin_runtime STATIC
  plugin_runtime/PluginLifecycleManager.cpp
)
add_library(Wyvern::PluginRuntime ALIAS wyvern_plugin_runtime)

target_compile_features(wyvern_plugin_runtime PUBLIC cxx_std_20)

target_include_directories(wyvern_plugin_runtime
  PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/..>
)
