cmake_minimum_required(VERSION 3.20)
project(WyvernCore LANGUAGES CXX)

# Core зависит от Boost.Signals2 (header-only) через managers/EventBus.h
if(NOT TARGET Boost::headers)
  find_package(Boost REQUIRED)
endif()

add_library(WyvernCore STATIC
  Core.cpp
)

target_compile_features(WyvernCore PUBLIC cxx_std_20)

target_include_directories(WyvernCore
  PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/..>
)

target_link_libraries(WyvernCore PUBLIC Boost::headers)
